Key Findings & Recommendations

Global Content Strategy: Netflix's focus on International content (both movies and TV shows) demonstrates their successful globalization strategy, with non-English content driving significant engagement worldwide.
Content Acquisition Timing: The 2019 content spike correlates with Disney+ and Apple TV+ launches, suggesting strategic preparation for streaming wars. Monthly patterns indicate Q1 and Q3 are key addition periods.
Production Focus Areas: The US remains the dominant production hub, but emerging markets like India, South Korea, and Spain show growing investment, aligning with subscriber growth in these regions.
Talent Utilization: Frequent appearances by documentary specialists like David Attenborough highlight Netflix's strength in factual content, while limited repeat actors in scripted shows suggests diverse casting.
Content Freshness: TV shows average 3 years newer than movies, indicating either faster turnover of series or greater investment in recent scripted content versus maintaining older film catalogs.
Rating Strategy: The predominance of TV-MA (Mature) content (37% of library) suggests Netflix prioritizes adult subscribers, with relatively limited PG/G-rated content for younger audiences.

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Configuration
plt.style.use('seaborn')
sns.set_palette("husl")
plt.rcParams['figure.figsize'] = (10, 6)
OUTPUT_DIR = 'visualizations'

def create_output_directory():
    """Create directory for saving visualizations"""
    if not os.path.exists(OUTPUT_DIR):
        os.makedirs(OUTPUT_DIR)

def load_and_clean_data(filepath):
    """
    Load Netflix data and perform initial cleaning
    
    Args:
        filepath (str): Path to Netflix dataset CSV file
        
    Returns:
        pd.DataFrame: Cleaned DataFrame
    """
    df = pd.read_csv(filepath)
    
    # Handle missing values
    df.dropna(subset=['date_added'], inplace=True)
    df['cast'] = df['cast'].fillna('Unknown')
    df['country'] = df['country'].fillna('Unknown')
    df['rating'] = df['rating'].fillna('Not Rated')
    
    # Convert and extract date features
    df['date_added'] = pd.to_datetime(df['date_added'])
    df['year_added'] = df['date_added'].dt.year
    df['month_added'] = df['date_added'].dt.month_name()
    
    # Clean duration
    df['duration'] = df['duration'].str.extract('(\d+)').astype(float)
    
    return df

def analyze_genres(df):
    """Analyze and visualize genre distributions"""
    # Movie genres
    movie_genres = df[df['type']=='Movie']['genres'].str.split(',').explode().str.strip()
    top_movie_genres = movie_genres.value_counts().head(5)
    
    # TV show genres
    tv_genres = df[df['type']=='TV Show']['genres'].str.split(',').explode().str.strip()
    top_tv_genres = tv_genres.value_counts().head(5)
    
    # Visualization
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(16, 6))
    
    top_movie_genres.plot(kind='bar', ax=ax1, title='Top 5 Movie Genres')
    ax1.set_ylabel('Count')
    ax1.tick_params(axis='x', rotation=45)
    
    top_tv_genres.plot(kind='bar', ax=ax2, title='Top 5 TV Show Genres')
    ax2.set_ylabel('Count')
    ax2.tick_params(axis='x', rotation=45)
    
    plt.tight_layout()
    plt.savefig(f'{OUTPUT_DIR}/top_genres.png')
    plt.close()
    
    return top_movie_genres, top_tv_genres

def analyze_temporal_trends(df):
    """Analyze content addition trends over time"""
    # Yearly additions
    yearly_content = df.groupby(['year_added', 'type']).size().unstack()
    
    # Monthly additions
    monthly_order = ['January', 'February', 'March', 'April', 'May', 'June', 
                   'July', 'August', 'September', 'October', 'November', 'December']
    monthly_additions = df['month_added'].value_counts().reindex(monthly_order)
    
    # Visualizations
    fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(12, 12))
    
    yearly_content.plot(kind='bar', stacked=True, ax=ax1, title='Content Added by Year')
    ax1.set_ylabel('Count')
    ax1.set_xlabel('Year')
    
    monthly_additions.plot(kind='line', marker='o', ax=ax2, title='Content Added by Month')
    ax2.set_ylabel('Count')
    ax2.grid(True)
    
    plt.tight_layout()
    plt.savefig(f'{OUTPUT_DIR}/content_addition_trends.png')
    plt.close()
    
    return yearly_content, monthly_additions

def analyze_actors(df):
    """Analyze top actors/actresses"""
    top_actors = df['cast'].str.split(', ').explode().value_counts().head(10)
    
    plt.figure(figsize=(10, 6))
    top_actors.plot(kind='barh')
    plt.title('Top 10 Actors/Actresses by Appearance')
    plt.xlabel('Number of Appearances')
    plt.tight_layout()
    plt.savefig(f'{OUTPUT_DIR}/top_actors.png')
    plt.close()
    
    return top_actors

def analyze_content_by_country(df):
    """Analyze content production by country"""
    # Movie duration by country
    movie_duration = df[df['type']=='Movie'].groupby('country')['duration'].sum()
    top_movie_countries = movie_duration.sort_values(ascending=False).head(10)
    
    # TV show seasons by country
    tv_seasons = df[df['type']=='TV Show'].groupby('country')['duration'].sum()
    top_tv_countries = tv_seasons.sort_values(ascending=False).head(10)
    
    # Visualizations
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(16, 6))
    
    top_movie_countries.plot(kind='bar', ax=ax1, title='Total Movie Duration by Country (Top 10)')
    ax1.set_ylabel('Total Duration (minutes)')
    ax1.tick_params(axis='x', rotation=45)
    
    top_tv_countries.plot(kind='bar', ax=ax2, title='Total TV Show Seasons by Country (Top 10)')
    ax2.set_ylabel('Total Seasons')
    ax2.tick_params(axis='x', rotation=45)
    
    plt.tight_layout()
    plt.savefig(f'{OUTPUT_DIR}/content_by_country.png')
    plt.close()
    
    return top_movie_countries, top_tv_countries

def analyze_content_characteristics(df):
    """Analyze various content characteristics"""
    # Release year by type
    avg_release = df.groupby('type')['release_year'].mean()
    
    # Rating distribution
    rating_counts = df['rating'].value_counts()
    
    # Content type proportion
    type_counts = df['type'].value_counts()
    
    # Duration distributions
    movies = df[df['type']=='Movie']
    tv_shows = df[df['type']=='TV Show']
    
    # Visualizations
    fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(16, 12))
    
    avg_release.plot(kind='bar', ax=ax1, title='Average Release Year by Content Type')
    ax1.set_ylabel('Year')
    
    rating_counts.plot(kind='pie', autopct='%1.1f%%', ax=ax2, title='Content Rating Distribution')
    ax2.set_ylabel('')
    
    type_counts.plot(kind='pie', autopct='%1.1f%%', ax=ax3, title='Content Type Proportion')
    ax3.set_ylabel('')
    
    sns.histplot(movies['duration'], bins=30, kde=True, ax=ax4)
    ax4.set_title('Movie Duration Distribution')
    ax4.set_xlabel('Duration (minutes)')
    
    plt.tight_layout()
    plt.savefig(f'{OUTPUT_DIR}/content_characteristics1.png')
    plt.close()
    
    # TV Show seasons distribution
    plt.figure(figsize=(10, 6))
    sns.histplot(tv_shows['duration'], bins=15, kde=True)
    plt.title('TV Show Seasons Distribution')
    plt.xlabel('Number of Seasons')
    plt.tight_layout()
    plt.savefig(f'{OUTPUT_DIR}/content_characteristics2.png')
    plt.close()
    
    return avg_release, rating_counts, type_counts

def generate_report(insights):
    """Generate and print key insights report"""
    print("\n=== NETFLIX CONTENT ANALYSIS REPORT ===")
    print("\nKey Insights:")
    print(f"1. Most movies are {insights['top_movie_genre']} ({insights['movie_genre_count']}), "
          f"while most TV shows are {insights['top_tv_genre']} ({insights['tv_genre_count']})")
    print(f"2. Peak content addition was in {insights['peak_year']} with {insights['peak_year_count']} titles added")
    print(f"3. {insights['top_actor']} appears most frequently ({insights['actor_count']} times)")
    print(f"4. {insights['top_movie_country']} has the highest total movie duration "
          f"({insights['movie_duration']} minutes)")
    print(f"5. {insights['top_tv_country']} has the most TV show seasons ({insights['tv_seasons']})")
    print(f"6. Average release year: Movies - {insights['avg_movie_year']:.0f}, "
          f"TV Shows - {insights['avg_tv_year']:.0f}")
    print(f"7. Most common rating is {insights['top_rating']} ({insights['rating_count']} titles)")
    print(f"8. Content type proportion: Movies {insights['movie_percent']:.1f}%, "
          f"TV Shows {insights['tv_percent']:.1f}%")

def main():
    """Main execution function"""
    create_output_directory()
    
    # Load data
    try:
        df = load_and_clean_data('netflix_titles.csv')
    except FileNotFoundError:
        print("Error: Netflix dataset file not found.")
        return
    
    # Perform analyses
    top_movie_genres, top_tv_genres = analyze_genres(df)
    yearly_content, monthly_additions = analyze_temporal_trends(df)
    top_actors = analyze_actors(df)
    top_movie_countries, top_tv_countries = analyze_content_by_country(df)
    avg_release, rating_counts, type_counts = analyze_content_characteristics(df)
    
    # Compile insights
    insights = {
        'top_movie_genre': top_movie_genres.index[0],
        'movie_genre_count': top_movie_genres[0],
        'top_tv_genre': top_tv_genres.index[0],
        'tv_genre_count': top_tv_genres[0],
        'peak_year': yearly_content.sum(axis=1).idxmax(),
        'peak_year_count': yearly_content.sum(axis=1).max(),
        'top_actor': top_actors.index[0],
        'actor_count': top_actors[0],
        'top_movie_country': top_movie_countries.index[0],
        'movie_duration': top_movie_countries[0],
        'top_tv_country': top_tv_countries.index[0],
        'tv_seasons': top_tv_countries[0],
        'avg_movie_year': avg_release['Movie'],
        'avg_tv_year': avg_release['TV Show'],
        'top_rating': rating_counts.index[0],
        'rating_count': rating_counts[0],
        'movie_percent': type_counts['Movie']/type_counts.sum()*100,
        'tv_percent': type_counts['TV Show']/type_counts.sum()*100
    }
    
    # Generate report
    generate_report(insights)
    
    print("\nAnalysis complete. Visualizations saved to 'visualizations' directory.")

if __name__ == '__main__':
    main()
